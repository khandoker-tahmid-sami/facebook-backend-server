      to,
      subject,
      html,
    });
  } catch (err) {
    throw new Error(`Email delivery failed: ${err.message}`);
  }
};

const crypto = require("crypto");
const sendEmail = require("../util/sendEmail");

const forgotPassword = async (email, db) => {
  // 1. Check if user exists
  const user = db.get("users").find({ email }).value();

  if (!user) {
    throw new Error("No account found with that email");
  }

  // 2. Generate a secure random token
  const resetToken = crypto.randomBytes(32).toString("hex");

  // 3. Set expiry to 1 hour from now
  const resetTokenExpiry = new Date(Date.now() + 60 * 60 * 1000).toISOString();

  // 4. Save token + expiry to user in db
  db.get("users")
    .find({ email })
    .assign({ resetToken, resetTokenExpiry })
    .write();

  // 5. Build reset link (this goes to your frontend)
  const resetLink = `${process.env.CLIENT_URL}/reset-password?token=${resetToken}`;

  // 6. Send the email
  await sendEmail(
    email,
    "Password Reset Request",
    `
      <h2>Password Reset</h2>
      <p>You requested a password reset. Click the link below to reset your password.</p>
      <p>This link expires in <strong>1 hour</strong>.</p>
      <a href="${resetLink}" style="
        background:#1877f2;
        color:white;
        padding:10px 20px;
        text-decoration:none;
        border-radius:5px;
        display:inline-block;
        margin-top:10px;
      ">Reset Password</a>
      <p>If you did not request this, ignore this email.</p>
    `
  );

  return { message: "Password reset link sent to your email" };
};

const resetPassword = async (token, newPassword, db) => {
  // 1. Find user by reset token
  const user = db.get("users").find({ resetToken: token }).value();

  if (!user) {
    throw new Error("Invalid or expired reset token");
  }

  // 2. Check if token is expired
  const isExpired = new Date() > new Date(user.resetTokenExpiry);

  if (isExpired) {
    throw new Error("Reset token has expired. Please request a new one.");
  }

  // 3. Hash the new password
  const hashedPassword = bcrypt.hashSync(newPassword, 8);

  // 4. Update password and clear the reset token fields
  db.get("users")
    .find({ resetToken: token })
    .assign({
      password: hashedPassword,
      resetToken: null,
      resetTokenExpiry: null,
    })
    .write();

  return { message: "Password reset successfully. You can now log in." };
};

***********************What is dotenv****************************


What is dotenv?
dotenv is a package that loads environment variables from a .env file into process.env so your code can read them.

The Problem It Solves
You have sensitive data like API keys, passwords, secret keys that you don't want to hardcode in your code:


// BAD — hardcoded, visible to everyone who sees the code
const resend = new Resend("re_abc123secretkey");
const jwtSecret = "mysecretkey123";
Instead you store them in a .env file:


RESEND_API_KEY=re_abc123secretkey
SECRET_KEY=mysecretkey123
Then in your code you just write:


// GOOD — the actual value is hidden in .env
const resend = new Resend(process.env.RESEND_API_KEY);
const jwtSecret = process.env.SECRET_KEY;
How It Works

require("dotenv").config();  // reads .env file → loads into process.env
After this line runs, everything in your .env file is available anywhere in your app via process.env.VARIABLE_NAME.

Why Order Matters
This is exactly your current bug. In main.js:


const customRouter = require("./router");  // line 2 — loads sendEmail.js
                                           // sendEmail.js reads process.env.RESEND_API_KEY
                                           // but .env hasn't loaded yet → undefined ❌
require("dotenv").config();                // line 5 — too late!
It should be:


require("dotenv").config();                // line 1 — .env loads first ✅
const customRouter = require("./router");  // now process.env.RESEND_API_KEY exists
The .env File Is Never Committed to Git
That's the whole point — you add .env to .gitignore so your secrets never get pushed to GitHub. Other developers on your team create their own .env file locally with their own keys.

Problem 2 — You Also Need to Actually Add the Key
Make sure your .env file has the real key:


RESEND_API_KEY=re_xxxxxxxxxxxxxxxx   ← paste your actual key from resend.com
Both fixes are needed — move dotenv to line 1 in main.js, and make sure the key exists in .env.


*************When the raw body is simple text in postman*********************

This is a Postman configuration issue, not a code bug. Your body is not being sent as JSON.

In Postman, check how you're sending the body:

Fix — Set Body to raw JSON
In Postman:

Click the Body tab
Select raw
On the right dropdown change Text → JSON
Like this:


Body → raw → JSON (dropdown on the right)
Then type your body:


{
    "email": "prinon96@gmail.com"
}
Why This Happens
When you select raw → JSON, Postman automatically sets the Content-Type: application/json header. Without that header, Express's express.json() middleware doesn't know the body is JSON, so it doesn't parse it — leaving req.body empty, which makes req.body.email undefined.

That's why you get "Please provide your email" even though you typed the email — Express literally can't see it.
